wzrost <- read.csv2("Wzrost.csv", header=FALSE)
colnames(wzrost) <- c("dane")

plot(density(wzrost$dane, width = 1, kernel = "gaussian"), main = "Zestawienie estymowanych gstoci z rn szerekoci pasma", xlab = "x", ylab = "gsto", col="red")
lines(density(wzrost$dane, kernel = "gaussian"), col="green")
lines(density(wzrost$dane, width = 10, kernel = "gaussian"), col="blue")
lines(density(wzrost$dane, width = 100, kernel = "gaussian"), col="black")
legend(x=192, y=0.065, legend = c("width=1", "width=3,488(default)", "width=10", "width=100"), fill = c("red", "green", "blue", "black"))

Z wykresw zauway mona i wraz ze wzrostem szerokoci pasma ronie stopie dopasowania estymatora gstoci do podanego jdra. Przy niskiej wartoci tego parametru gsto za cile dopasowuje si do danych, a przy wysokiej wartoci width estymator zaczyna przypomina wybran funkcj jdrow przy minimalnym uwzgldnieniu podanych danych.

Na podstawie wykresw, mona wywnioskowa, e wybr jdra odpowiada za cz doboru estymatora gstoci niezalen od podanych danych. Wykresy dla duych szeregoci pasma rzeczywicie przybliaj podan funkcj jdrow, a przy mniejszych wartociach stanowi mieszank funkcji gstoci.

mu = mean(wzrost$dane); sigma = sd(wzrost$dane)
x = seq(120, 230, length = 500)
hist(wzrost$dane, 15, prob = T, main = "Zestawienie histogramy, estymowanej gstoci i rozkadu normalnego", xlab = "x", ylab = "gsto", col="magenta")
lines(density(wzrost$dane, kernel = "epanechnikov"), lwd = 2, col = "cyan")
lines(x, dnorm(x, mu, sigma), lwd = 2, col = "gold")
legend("topright", legend = c("histogram", "estymowana gsto", "rozkad normalny"), fill = c("magenta", "cyan", "gold"))
